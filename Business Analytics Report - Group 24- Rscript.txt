# Installing packages
require(tidyverse)
theme_minimal(0)

# To read the data
data <- read.csv("slump_test.csv")
View(data)

# To check the  data for duplicates
str(data)
colSums(is.na(data))
duplicates <- duplicated(data)
data[duplicates, ]

# Deleting duplicates
data_unique <- data[!duplicated(data), ]

# Boxplot for outlier
boxplot(data_unique, 
        col = rainbow(ncol(data_unique)),  # Assigns a different color to each box
        main = "Boxplot of Slump data")


# Calculating correlation matrix
cor_matrix <- cor(data_unique)
print(cor_matrix)

# Pair plotting individually for checking linearity based upon correlation

# Pair 1
ggplot(data_unique, aes(x = strength, y = cement)) +
  geom_point() +
  geom_smooth(method = "lm", se = FALSE, color = "blue")+
  labs(title = "Strength vs Cement", x = "Strength", y = "Cement") +
  theme_minimal() +
  theme(
    panel.grid = element_blank(),

# Descriptive statistics
summary(data_unique)

# Regression analysis

# Model 1
model1 <- lm(strength ~ water+cement+fly_ash,data=data_unique)
summary(model1)
par(mfrow=c(2, 2))
plot(model1)

# Model 2
model2 <- lm(strength ~ cement+fly_ash,data=data_unique)
summary(model2)
par(mfrow=c(2, 2))
plot(model2)


# Model 3
model3<-lm(strength~slag+water+coarse_aggr+fine_aggr+fly_ash,data=data_unique)
summary(model3)
par(mfrow=c(2, 2))
plot(model3)

# Analzying the fit

# Is model 3 the variable normally distributed?
hist(residuals(model3))
library(lmtest)
bptest(model3)
shapiro.test(model3$residuals)

# Transforming the model with square root function : Model_sqrt
model_sqrt <- lm(sqrt(strength) ~ slag + water + coarse_aggr + fine_aggr+fly_ash, data = data_unique)
summary(model_sqrt)
par(mfrow=c(2, 2))
plot(model_sqrt)

# Again testing the assumptions
hist(residuals(model_sqrt))
library(lmtest)
bptest(model_sqrt)
shapiro.test(model_sqrt$residuals)
